## ?? EXPORTAR A EXCEL

```csharp
public void ExportarExcel(DataTable dt, string nombreArchivo)
{
    System.IO.StreamWriter sw = new System.IO.StreamWriter(nombreArchivo + ".csv");
    
    // Escribir encabezados
    for (int i = 0; i < dt.Columns.Count; i++)
    {
        sw.Write(dt.Columns[i].ColumnName + "\t");
    }
    sw.WriteLine();
    
    // Escribir datos
    foreach (DataRow dr in dt.Rows)
    {
        for (int i = 0; i < dt.Columns.Count; i++)
        {
            sw.Write(dr[i].ToString() + "\t");
        }
        sw.WriteLine();
    }
    
    sw.Close();
}
```
